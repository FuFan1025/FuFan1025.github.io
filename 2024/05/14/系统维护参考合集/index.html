# 查看当前计算机的初始安装日期和系统启动时间

systeminfo |findstr -i "初始安装日期 系统启动时间"

# 查看系统共享服务

net share

net share 服务名 #开启服务共享

net share 服务名 #删除开启的共享服务

# 关机指令

shutdown /s /t "times"
  • /s 关机指令
  • /t 设置关机时间 例：3600 为一个小时
  • /a 取消关机

# 生成目录树结构

tree "文件夹路径" > xxx.txt

# 拷贝文件

copy text.txt c:\doc\key_bak.txt

# 查找文件中 ok

find "ok" '文件路径' && echo "成功" || echo "不成功"

# 查看网络连接以及每一个网络接口设备状态

netstat -an

# 将工作站计算机配置【指定域名】和 192.168.0.138 这个 IP 绑定

  1. 打开 C:\Windows\System32\drivers\etc\hosts
  2. 在文件的末尾添加一行，格式为 “IP 地址 域名”，例如 “192.168.0.138 指定域名”
